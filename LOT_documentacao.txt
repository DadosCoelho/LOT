  function sleep(ms) { return new Promise(r => setTimeout(r, ms)); }

  function formatMoney(v) {
    return v.toLocaleString("pt-BR", { minimumFractionDigits: 2, maximumFractionDigits: 2 });
  }

  function formatarData(dataStr) {
    if (!dataStr) return "Sem data";
    try {
      let d;
      if (dataStr.includes('-')) d = new Date(dataStr);
      else if (dataStr.includes('/')) { const p = dataStr.split('/'); d = new Date(p[2], p[1]-1, p[0]); }
      else d = new Date(dataStr);
      if (isNaN(d.getTime())) return dataStr;
      return `${String(d.getDate()).padStart(2,'0')}/${String(d.getMonth()+1).padStart(2,'0')}/${d.getFullYear()}`;
    } catch(e) { return dataStr; }
  }

  function calcularAcertos(dezenasSorteadas, minhasDezenas) {
    return minhasDezenas.filter(d => dezenasSorteadas.includes(d)).length;
  }

  // Faixas Dupla Sena:  1–4 = 1º sorteio (sena/quina/quadra/terno),  5–8 = 2º sorteio
  function buscarPremioDuplaSena(resultado, acertos, sorteio) {
    if (!resultado.listaRateioPremio || acertos < 3) return 0;
    const faixasMap = {
      1: { 6:1, 5:2, 4:3, 3:4 },
      2: { 6:5, 5:6, 4:7, 3:8 }
    };
    const faixa = faixasMap[sorteio][acertos];
    if (!faixa) return 0;

    // Por número de faixa
    const it = resultado.listaRateioPremio.find(i => i.faixa === faixa);
    if (it && typeof it.valorPremio === "number") return it.valorPremio;

    // Fallback por descrição
    const labelAcertos = { 6:"sena", 5:"quina", 4:"quadra", 3:"terno" }[acertos] || "";
    for (const item of resultado.listaRateioPremio) {
      const desc = (item.descricaoFaixa || item.descricao || "").toLowerCase();
      if (desc.includes(labelAcertos) && desc.includes(String(sorteio))) {
        return typeof item.valorPremio === "number" ? item.valorPremio : 0;
      }
    }
    return 0;
  }

  function distribuirPorJogador(premioTotal) {
    const totalCotas = JOGADORES.reduce((s, j) => s + j.cotas, 0);
    if (!totalCotas || !premioTotal) return JOGADORES.map(j => ({ ...j, valor: 0 }));
    const porCota = premioTotal / totalCotas;
    return JOGADORES.map(j => ({ ...j, valor: porCota * j.cotas }));
  }

  function renderizarResultadoJogo(jogo, draw1, draw2, prize1, prize2) {
    const totalJogo  = prize1 + prize2;
    const acertos1   = calcularAcertos(draw1, jogo.dezenas);

  async function buscarDadosConcurso() {
    const url = CONCURSO_PASCOA
      ? `https://api.guidi.dev.br/loteria/duplasena/${CONCURSO_PASCOA}`
      : `https://api.guidi.dev.br/loteria/duplasena/ultimo`;
    for (let t = 0; t < 3; t++) {
      try {
        if (t > 0) await sleep(2000 * t);
        const resp = await fetch(url);
        if (!resp.ok) {
          if (resp.status === 404) {
            return { status: "notfound" };
          }
          throw new Error(`HTTP ${resp.status}`);
        }
        return await resp.json();
      } catch(e) {
        if (t === 2) throw e;
      }
    }
  }

  // ── Fluxo principal ──

  async function conferirTodos() {
    loadingEl.style.display = "block";
    criarCardResumoGlobal(0, 0, JOGADORES.map(j => ({ ...j, valor: 0 })));
    JOGOS.forEach(j => criarCardBase(j));

    let dados = null;
    try {
      dados = await buscarDadosConcurso();
    } catch(e) {
      JOGOS.forEach(j => {
        setStatusCard(j.id, "erro", `⚠️ Erro ao buscar dados: ${e.message}`);
        const msg = document.getElementById(`loading-msg-${j.id}`);
        if (msg) { msg.textContent = "⚠️ Não foi possível obter os dados do sorteio"; msg.style.background = "#fee2e2"; msg.style.color = "#991b1b"; }
      });
      loadingEl.style.display = "none";
      return;
    }

    loadingEl.style.display = "none";

    // Verifica se o sorteio já foi realizado (Dupla Sena tem 2 sorteios = listaDezenas com 2 arrays)
    const listaOk =
        dados.listaDezenas &&
        (
            dados.listaDezenasSegundoSorteio || // novo formato
            (Array.isArray(dados.listaDezenas[0]) && dados.listaDezenas.length >= 2) ||
            dados.listaDezenas.length >= 12
        );

    if (!listaOk || dados?.status === "notfound") {
      const concurso = dados?.numero || CONCURSO_PASCOA || "—";
      const data     = dados?.dataApuracao ? formatarData(dados.dataApuracao) : null;
      JOGOS.forEach(j => {
        setStatusCard(j.id, "aguardando",
          `⏸️ <strong>Aguardando sorteio</strong> · Concurso ${concurso}${data ? ` · Previsto: ${data}` : ""}`);
        const msg = document.getElementById(`loading-msg-${j.id}`);
        if (msg) {
          msg.textContent = "⏸️ Sorteio ainda não realizado. Volte após o sorteio!";
          msg.style.background = "#fef3c7";
          msg.style.color = "#92400e";
        }
      });
      return;
    }

    // Extrai os 2 sorteios
    let draw1 = [];
    let draw2 = [];

    if (dados.listaDezenasSegundoSorteio) {
    // ✅ NOVO formato da API (correto)
    draw1 = dados.listaDezenas;
    draw2 = dados.listaDezenasSegundoSorteio;

    } else if (Array.isArray(dados.listaDezenas[0])) {
    // formato antigo (2 arrays)
    draw1 = dados.listaDezenas[0];
    draw2 = dados.listaDezenas[1];

    } else {
    // fallback (12 números em 1 array)
    draw1 = dados.listaDezenas.slice(0, 6);
    draw2 = dados.listaDezenas.slice(6, 12);
    }

    const concurso = dados.numero || "—";
    const dataStr  = formatarData(dados.dataApuracao);

    // Calcula e renderiza cada jogo
    let totalGeral    = 0;
    let jogosComPremio = 0;
    let distribGlobal  = JOGADORES.map(j => ({ ...j, valor: 0 }));

    for (const jogo of JOGOS) {
      const prize1 = buscarPremioDuplaSena(dados, calcularAcertos(draw1, jogo.dezenas), 1);
      const prize2 = buscarPremioDuplaSena(dados, calcularAcertos(draw2, jogo.dezenas), 2);
      const total  = prize1 + prize2;

      if (total > 0) jogosComPremio++;
      totalGeral += total;

      const distrib = distribuirPorJogador(total);
      distribGlobal = distribGlobal.map((g, i) => ({ ...g, valor: g.valor + distrib[i].valor }));

      setStatusCard(jogo.id, "realizado",
        `✅ <strong>Sorteio realizado!</strong> · Concurso ${concurso} · Data: ${dataStr}`);

      renderizarResultadoJogo(jogo, draw1, draw2, prize1, prize2);
    }

    criarCardResumoGlobal(totalGeral, jogosComPremio, distribGlobal);
  }

  function sleep(ms) { return new Promise(r => setTimeout(r, ms)); }

  function formatMoney(v) {
    return v.toLocaleString("pt-BR", { minimumFractionDigits: 2, maximumFractionDigits: 2 });
  }

  function formatarData(dataStr) {
    if (!dataStr) return "Sem data";
    try {
      let d;
      if (dataStr.includes('-')) d = new Date(dataStr);
      else if (dataStr.includes('/')) {
        const p = dataStr.split('/');
        d = new Date(p[2], p[1] - 1, p[0]);
      } else d = new Date(dataStr);
      if (isNaN(d.getTime())) return dataStr;
      return `${String(d.getDate()).padStart(2,'0')}/${String(d.getMonth()+1).padStart(2,'0')}/${d.getFullYear()}`;
    } catch(e) { return dataStr; }
  }

  async function buscarConcursoAtual(modalidade) {
    try {
      const resp = await fetch(`https://api.guidi.dev.br/loteria/${modalidade}/ultimo`);
      if (!resp.ok) throw new Error(`HTTP ${resp.status}`);
      const data = await resp.json();
      return data.numero || null;
    } catch(e) { console.error('Erro ao buscar concurso atual:', e); return null; }
  }

  function calcularAcertos(dezenasSorteadas, minhasDezenas) {
    return minhasDezenas.filter(d => dezenasSorteadas.includes(d)).length;
  }

  function buscarPremioLotoFacil(resultado, acertos) {
    if (!resultado.listaRateioPremio || acertos < 11 || acertos > 15) return 0;
    for (const item of resultado.listaRateioPremio) {
      const desc = (item.descricaoFaixa || item.descricao || '').toString().toLowerCase();
      if (desc.includes(`${acertos} acerto`)) return typeof item.valorPremio === "number" ? item.valorPremio : 0;
    }
    const faixaMap = { 15:1, 14:2, 13:3, 12:4, 11:5 };
    const faixaEsperada = faixaMap[acertos];
    if (faixaEsperada) {
      const it = resultado.listaRateioPremio.find(i => i.faixa === faixaEsperada);
      if (it && typeof it.valorPremio === "number") return it.valorPremio;
    }
    const idx = 15 - acertos;
    if (idx >= 0 && idx < resultado.listaRateioPremio.length) {
      const it = resultado.listaRateioPremio[idx];
      if (it && typeof it.valorPremio === "number") return it.valorPremio;
    }
    return 0;
  }

  function buscarPremioGenerico(resultado, acertos, modalidade) {
    if (!resultado.listaRateioPremio) return 0;
    if (modalidade === "lotofacil") return buscarPremioLotoFacil(resultado, acertos);

  async function conferirJogo(jogo) {
    let totalPremios = 0, concursosComPremio = 0, melhorAcerto = 0, concursosConferidos = 0;
    const MAX_TENTATIVAS = 3;
    let jogadoresAcumulado = jogo.jogadores.map(j => ({ ...j, valor: 0 }));
    concursosDetalhes[jogo.id] = [];

    const concursoAtual = await buscarConcursoAtual(jogo.modalidade);
    const concursoFinal = jogo.concursoInicial + jogo.quantidadeTeimosinha - 1;
    let ultimoConcurso  = concursoFinal;
    if (concursoAtual !== null && concursoAtual < concursoFinal) ultimoConcurso = concursoAtual;
    if (concursoAtual !== null && concursoAtual < jogo.concursoInicial) ultimoConcurso = jogo.concursoInicial - 1;

    const qtde = Math.max(0, ultimoConcurso - jogo.concursoInicial + 1);
    atualizarStatusTeimosinha(jogo.id, concursoAtual, jogo, 0);

    if (qtde === 0) {
      const msg = document.getElementById(`loading-msg-${jogo.id}`);
      if (msg) { msg.textContent = '⏸️ Nenhum concurso disponível ainda'; msg.style.background = '#fef3c7'; }
      mostrarResumoEValores(jogo.id);
      return;
    }

    for (let i = qtde - 1; i >= 0; i--) {
      const numeroConcurso = jogo.concursoInicial + i;
      const isUltimo = (i === qtde - 1);
      let sucesso = false;

      for (let t = 0; t < MAX_TENTATIVAS && !sucesso; t++) {
        try {
          if (t > 0) await sleep(2000 * t);
          const resp = await fetch(`https://api.guidi.dev.br/loteria/${jogo.modalidade}/${numeroConcurso}`);
          if (!resp.ok) throw new Error(`HTTP ${resp.status}`);
          const data = await resp.json();

          if (!data.listaDezenas) {
            adicionarDetalheConcurso(jogo.id, {
              numero: numeroConcurso, acertos: 0, premioTotal: 0,
              dezenasSorteadas: [], dezenasDoJogo: jogo.dezenas, dataApuracao: "Sem dados"
            }, isUltimo);
            concursosConferidos++; sucesso = true; continue;
          }

          const dezenasSorteadas = Array.isArray(data.listaDezenas[0]) ? data.listaDezenas[0] : data.listaDezenas;
          const acertos          = calcularAcertos(dezenasSorteadas, jogo.dezenas);
          const premio           = buscarPremioGenerico(data, acertos, jogo.modalidade);

          if (acertos > melhorAcerto) melhorAcerto = acertos;
          if (premio > 0 && acertos >= 11) concursosComPremio++;
          totalPremios += premio;

          const distrib = distribuirPorJogador(premio, jogo.jogadores);
          jogadoresAcumulado = jogadoresAcumulado.map((j, idx) => ({ ...j, valor: j.valor + distrib[idx].valor }));

          adicionarDetalheConcurso(jogo.id, {
            numero: numeroConcurso, acertos, premioTotal: premio,
            dezenasSorteadas, dezenasDoJogo: jogo.dezenas,
            dataApuracao: formatarData(data.dataApuracao)
          }, isUltimo);

          concursosConferidos++; sucesso = true;
        } catch(e) {
          if (t === MAX_TENTATIVAS - 1) {
            const details = document.getElementById(`details-${jogo.id}`);
            if (details) {
              const err = document.createElement("div");
              err.className = "error";
              err.textContent = `Erro no concurso ${numeroConcurso}: ${e.message}`;
              details.appendChild(err);
            }
          }
        }
      }
      await sleep(800 + Math.random() * 400);
    }

    atualizarResumo(jogo.id, { totalPremios, concursosComPremio, melhorAcerto, concursosConferidos });
    atualizarJogadoresTabela(jogo.id, jogadoresAcumulado);
    atualizarStatusTeimosinha(jogo.id, concursoAtual, jogo, concursosConferidos);
    mostrarResumoEValores(jogo.id);
  }
